interface IAdminUser extends Document {

- **isActive**: Boolean, default true
- **lastLogin**: Optional Date for tracking user activity

**Indexes:**
- `{ email: 1 }` (unique)
- `{ role: 1 }`
- `{ isActive: 1 }`
